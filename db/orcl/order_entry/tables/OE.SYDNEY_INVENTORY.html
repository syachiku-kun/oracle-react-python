CREATE OR REPLACE FORCE EDITIONABLE VIEW "OE"."SYDNEY_INVENTORY" ("PRODUCT_ID", "PRODUCT_NAME", "QUANTITY_ON_HAND") AS 
  SELECT p.product_id
,      p.product_name
,      i.quantity_on_hand
FROM   inventories i
,      warehouses  w
,      products    p
WHERE  p.product_id = i.product_id
AND    i.warehouse_id = w.warehouse_id
AND    w.warehouse_name = 'Sydney'
